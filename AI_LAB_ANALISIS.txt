import os
import json
from datetime import datetime


# =========================================================
# SYNERGIA CORE NEXT PRO

BASE_DIR = os.path.dirname(os.path.abspath(__file__))

CONVERSATIONS_PATH = os.path.join(
    BASE_DIR,
    "conversations"
)

SESSIONS_PATH = os.path.join(
    BASE_DIR,
    "sessions"
)

PROMPTS_PATH = os.path.join(
    BASE_DIR,
    "prompts"
)


class MemoryManager:

    # =====================================================
    # INIT
    # =====================================================

    def __init__(self):

        self.ensure_directories()

    # =====================================================
    # CREATE DIRECTORIES
    # =====================================================

    def ensure_directories(self):

        os.makedirs(CONVERSATIONS_PATH, exist_ok=True)
        os.makedirs(SESSIONS_PATH, exist_ok=True)
        os.makedirs(PROMPTS_PATH, exist_ok=True)

    # =====================================================
    # SESSION FILE PATH
    # =====================================================

    def get_session_filepath(self, session_id):

        filename = f"{session_id}.json"

        return os.path.join(
            CONVERSATIONS_PATH,
            filename
        )

    # =====================================================
    # CREATE SESSION
    # =====================================================

    def create_session(self, session_id):

        filepath = self.get_session_filepath(session_id)

        if not os.path.exists(filepath):

            with open(filepath, "w", encoding="utf-8") as f:

                json.dump(
                    [],
                    f,
                    indent=4,
                    ensure_ascii=False
                )

            print(f"[SESSION CREATED] {session_id}")

        return filepath

    # =====================================================
    # SAVE INTERACTION
    # =====================================================

    def save_interaction(
        self,
        session_id,
        model,
        prompt,
        response
    ):

        self.create_session(session_id)

        filepath = self.get_session_filepath(session_id)

        interaction = {

            "timestamp": datetime.now().strftime(
                "%Y-%m-%d %H:%M:%S"
            ),

            "session_id": session_id,

            "model": model,

            "prompt": prompt,

            "response": response
        }

        history = []

        # =============================================
        # LOAD OLD HISTORY
        # =============================================

        if os.path.exists(filepath):

            try:

                with open(filepath, "r", encoding="utf-8") as f:

                    history = json.load(f)

            except Exception as e:

                print("[MEMORY LOAD ERROR]")
                print(e)

                history = []

        # =============================================
        # APPEND NEW INTERACTION
        # =============================================

        history.append(interaction)

        # =============================================
        # SAVE UPDATED HISTORY
        # =============================================

        try:

            with open(filepath, "w", encoding="utf-8") as f:

                json.dump(
                    history,
                    f,
                    indent=4,
                    ensure_ascii=False
                )

            print(f"[MEMORY SAVED] {filepath}")

        except Exception as e:

            print("[MEMORY SAVE ERROR]")
            print(e)

    # =====================================================
    # LOAD CONVERSATION
    # =====================================================

    def load_conversation(self, session_id):

        filepath = self.get_session_filepath(session_id)

        if not os.path.exists(filepath):

            print("[SESSION NOT FOUND]")

            return []

        try:

            with open(filepath, "r", encoding="utf-8") as f:

                return json.load(f)

        except Exception as e:

            print("[LOAD CONVERSATION ERROR]")
            print(e)

            return []

    # =====================================================
    # LIST SESSIONS
    # =====================================================

    def list_sessions(self):

        self.ensure_directories()

        sessions = []

        for file in os.listdir(CONVERSATIONS_PATH):

            if file.endswith(".json"):

                sessions.append(
                    file.replace(".json", "")
                )

        sessions.sort()

        return sessions

    # =====================================================
    # GET LAST MESSAGES
    # =====================================================

    def get_last_messages(
        self,
        session_id,
        limit=5
    ):

        history = self.load_conversation(session_id)

        if not history:
            return []

        return history[-limit:]

    # =====================================================
    # DELETE SESSION
    # =====================================================

    def delete_session(self, session_id):

        filepath = self.get_session_filepath(session_id)

        if not os.path.exists(filepath):

            print("[SESSION DOES NOT EXIST]")

            return

        try:

            os.remove(filepath)

            print(f"[SESSION DELETED] {session_id}")

        except Exception as e:

            print("[DELETE SESSION ERROR]")
            print(e)

    # =====================================================
    # SEARCH IN CONVERSATION
    # =====================================================

    def search_in_conversation(
        self,
        session_id,
        keyword
    ):

        history = self.load_conversation(session_id)

        if not history:
            return []

        keyword = keyword.lower()

        results = []

        for item in history:

            prompt = item.get(
                "prompt",
                ""
            ).lower()

            response = item.get(
                "response",
                ""
            ).lower()

            if keyword in prompt or keyword in response:

                results.append(item)

        return results

    # =====================================================
    # EXPORT SESSION TXT
    # =====================================================

    def export_session_txt(
        self,
        session_id
    ):

        history = self.load_conversation(session_id)

        if not history:

            print("[NO HISTORY FOUND]")

            return

        export_path = os.path.join(
            SESSIONS_PATH,
            f"{session_id}.txt"
        )

        try:

            with open(export_path, "w", encoding="utf-8") as f:

                f.write(
                    "=====================================\n"
                )

                f.write(
                    "SYNERGIA SESSION EXPORT\n"
                )

                f.write(
                    "=====================================\n\n"
                )

                for item in history:

                    f.write(
                        f"TIMESTAMP: {item['timestamp']}\n"
                    )

                    f.write(
                        f"MODEL: {item['model']}\n"
                    )

                    f.write(
                        f"PROMPT:\n{item['prompt']}\n\n"
                    )

                    f.write(
                        f"RESPONSE:\n{item['response']}\n"
                    )

                    f.write(
                        "\n-------------------------------------\n\n"
                    )

            print(f"[SESSION EXPORTED] {export_path}")

        except Exception as e:

            print("[EXPORT ERROR]")
            print(e)

    # =====================================================
    # CLEAR CONVERSATION
    # =====================================================

    def clear_conversation(self, session_id):

        filepath = self.get_session_filepath(session_id)

        if not os.path.exists(filepath):

            print("[SESSION NOT FOUND]")

            return

        try:

            with open(filepath, "w", encoding="utf-8") as f:

                json.dump(
                    [],
                    f,
                    indent=4,
                    ensure_ascii=False
                )

            print(f"[CONVERSATION CLEARED] {session_id}")

        except Exception as e:

            print("[CLEAR CONVERSATION ERROR]")
            print(e)

    # =====================================================
    # GET SESSION STATS
    # =====================================================

    def get_session_stats(self, session_id):

        history = self.load_conversation(session_id)

        if not history:

            return {
                "messages": 0,
                "models": [],
                "last_timestamp": None
            }

        models = list(set(
            item["model"]
            for item in history
        ))

        return {

            "messages": len(history),

            "models": models,

            "last_timestamp": history[-1]["timestamp"]
        }

class OllamaProvider:

    # =====================================================
    # INIT
    # =====================================================

    def __init__(self):

        print("[OLLAMA PROVIDER LOADED]")

    # =====================================================
    # SIMPLE GENERATE
    # =====================================================

    def generate(
        self,
        model_name,
        prompt,
        session_id="default_session"
    ):

        try:

            response = ollama.chat(

                model=model_name,

                messages=[
                    {
                        "role": "user",
                        "content": prompt
                    }
                ]
            )

            content = response["message"]["content"]

            # =============================================
            # SAVE MEMORY
            # =============================================

            memory.save_interaction(

                session_id=session_id,

                model=model_name,

                prompt=prompt,

                response=content
            )

            return content

        except Exception as e:

            print("[OLLAMA GENERATE ERROR]")
            print(e)

            return None

    # =====================================================
    # STREAM GENERATE
    # =====================================================

    def generate_stream(

        self,

        model_name,

        prompt,

        session_id="default_session"
    ):

        full_response = ""

        try:

            stream = ollama.chat(

                model=model_name,

                messages=[
                    {
                        "role": "user",
                        "content": prompt
                    }
                ],

                stream=True
            )

            for chunk in stream:

                token = chunk["message"]["content"]

                print(token, end="", flush=True)

                full_response += token

            print("\n")

            # =============================================
            # SAVE MEMORY
            # =============================================

            memory.save_interaction(

                session_id=session_id,

                model=model_name,

                prompt=prompt,

                response=full_response
            )

            return full_response

        except Exception as e:

            print("[OLLAMA STREAM ERROR]")
            print(e)

            return None

    # =====================================================
    # GENERATE WITH CONTEXT
    # =====================================================

    def generate_with_context(

        self,

        model_name,

        prompt,

        session_id="default_session",

        context_limit=5
    ):

        try:

            history = memory.get_last_messages(

                session_id=session_id,

                limit=context_limit
            )

            messages = []

            # =============================================
            # LOAD HISTORY
            # =============================================

            for item in history:

                messages.append(
                    {
                        "role": "user",
                        "content": item["prompt"]
                    }
                )

                messages.append(
                    {
                        "role": "assistant",
                        "content": item["response"]
                    }
                )

            # =============================================
            # NEW PROMPT
            # =============================================

            messages.append(
                {
                    "role": "user",
                    "content": prompt
                }
            )

            response = ollama.chat(

                model=model_name,

                messages=messages
            )

            content = response["message"]["content"]

            # =============================================
            # SAVE MEMORY
            # =============================================

            memory.save_interaction(

                session_id=session_id,

                model=model_name,

                prompt=prompt,

                response=content
            )

            return content

        except Exception as e:

            print("[OLLAMA CONTEXT ERROR]")
            print(e)

            return None

import requests
import json


class OllamaConnector:

    def __init__(self):

        self.url = "http://localhost:11434/api/generate"

    def generate_stream(self, model, prompt):

        payload = {
            "model": model,
            "prompt": prompt,
            "stream": True
        }

        try:

            response = requests.post(
                self.url,
                json=payload,
                stream=True
            )

            for line in response.iter_lines():

                if line:

                    data = json.loads(line)

                    token = data.get("response", "")

                    yield token

        except Exception as e:

            yield f"ERROR: {e}"

        selected = self.models_list.currentItem()

        if not selected:

            self.output.setText(
                "SeleccionГЎ un modelo."
            )

            return

        model = selected.text()

        prompt = self.prompt_input.toPlainText()

        self.output.clear()

        self.worker = StreamWorker(
            model,
            prompt
        )

        self.worker.token_received.connect(
            self.update_output
        )

        self.worker.start()

    def update_output(self, token):

        self.output.insertPlainText(token)

        scrollbar = self.output.verticalScrollBar()

        scrollbar.setValue(
            scrollbar.maximum()
        )
